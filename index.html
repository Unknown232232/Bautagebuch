// PDF-Export Funktionen f√ºr das Frontend // Diese Funktionen in die bestehende JavaScript-Datei oder in index.html einbinden // Komplettes Bautagebuch als PDF exportieren function exportPDF() { showLoading('PDF wird erstellt...'); fetch('/api/export/pdf') .then(response => { if (!response.ok) { throw new Error('PDF-Export fehlgeschlagen'); } return response.blob(); }) .then(blob => { // PDF Download const url = window.URL.createObjectURL(blob); const a = document.createElement('a'); a.style.display = 'none'; a.href = url; a.download = `Bautagebuch_${new Date().toISOString().split('T')[0]}.pdf`; document.body.appendChild(a); a.click(); window.URL.revokeObjectURL(url); hideLoading(); showAlert('PDF erfolgreich erstellt!', 'success'); }) .catch(error => { hideLoading(); showAlert('Fehler beim PDF-Export: ' + error.message, 'error'); console.error('PDF Export Error:', error); }); } // Einzelnen Eintrag als PDF exportieren function exportEntryPDF(entryId) { showLoading('PDF wird erstellt...'); fetch(`/api/export/pdf/entry/${entryId}`) .then(response => { if (!response.ok) { throw new Error('PDF-Export fehlgeschlagen'); } return response.blob(); }) .then(blob => { // PDF Download const url = window.URL.createObjectURL(blob); const a = document.createElement('a'); a.style.display = 'none'; a.href = url; a.download = `Eintrag_${entryId}.pdf`; document.body.appendChild(a); a.click(); window.URL.revokeObjectURL(url); hideLoading(); showAlert('PDF erfolgreich erstellt!', 'success'); }) .catch(error => { hideLoading(); showAlert('Fehler beim PDF-Export: ' + error.message, 'error'); console.error('PDF Export Error:', error); }); } // Loading-Anzeige Funktionen function showLoading(message = 'Lade...') { let loadingDiv = document.getElementById('loading-overlay'); if (!loadingDiv) { loadingDiv = document.createElement('div'); loadingDiv.id = 'loading-overlay'; loadingDiv.innerHTML = `
